describe('Phase 15: Reconnect After Disconnect', () => {
  // Helper: create a 3-player lobby (Alice host, Bob, Carol) in playing state
  function makeReconnectLobby(roomCode = 'RECON01') {
    lobbies.delete(roomCode);
    createLobby(roomCode, 'host-socket', 'Alice');
    addPlayer(roomCode, 'p2-socket', 'Bob');
    addPlayer(roomCode, 'p3-socket', 'Carol');
    setSelectedPuzzle(roomCode, 'puzzle_01');
    const result = startGame(roomCode);
    if (!result.ok) throw new Error('startGame failed: ' + result.error);
    return lobbies.get(roomCode);
  }

  // Cleanup helper — clears any pending timers on all players
  function cleanupTimers(lobby) {
    if (!lobby) return;
    lobby.players.forEach(p => {
      if (p.disconnectTimer) { clearTimeout(p.disconnectTimer); p.disconnectTimer = null; }
    });
  }

  describe('reservePlayerSlot', () => {
    it('marks player as disconnected and returns ok with playerName', () => {
      const lobby = makeReconnectLobby('RSV01');
      const result = reservePlayerSlot('RSV01', 'p2-socket', () => {});
      assert.equal(result.ok, true);
      assert.equal(result.playerName, 'Bob');
      const bob = lobby.players.find(p => p.name === 'Bob');
      assert.equal(bob.disconnected, true);
      assert.ok(bob.disconnectTimer !== null);
      cleanupTimers(lobby);
      lobbies.delete('RSV01');
    });

    it('advances turn when active player disconnects', () => {
      const lobby = makeReconnectLobby('RSV02');
      // Alice (index 0) is active by default
      assert.equal(lobby.activeTurnIndex, 0);
      reservePlayerSlot('RSV02', 'host-socket', () => {});
      // Should advance past Alice (disconnected) to Bob (index 1)
      assert.equal(lobby.activeTurnIndex, 1);
      cleanupTimers(lobby);
      lobbies.delete('RSV02');
    });

    it('does NOT advance turn when non-active player disconnects', () => {
      const lobby = makeReconnectLobby('RSV03');
      assert.equal(lobby.activeTurnIndex, 0); // Alice active
      reservePlayerSlot('RSV03', 'p2-socket', () => {}); // Bob disconnects (not active)
      assert.equal(lobby.activeTurnIndex, 0); // Still Alice
      cleanupTimers(lobby);
      lobbies.delete('RSV03');
    });

    it('returns ok:false for unknown socketId', () => {
      makeReconnectLobby('RSV04');
      const result = reservePlayerSlot('RSV04', 'nonexistent', () => {});
      assert.equal(result.ok, false);
      lobbies.delete('RSV04');
    });
  });

  describe('reconnectPlayer', () => {
    it('re-associates socketId, clears disconnected flag and timer', () => {
      const lobby = makeReconnectLobby('RCN01');
      reservePlayerSlot('RCN01', 'p2-socket', () => {});
      const result = reconnectPlayer('RCN01', 'Bob', 'new-p2-socket');
      assert.equal(result.ok, true);
      const bob = lobby.players.find(p => p.name === 'Bob');
      assert.equal(bob.socketId, 'new-p2-socket');
      assert.equal(bob.disconnected, false);
      assert.equal(bob.disconnectTimer, null);
      cleanupTimers(lobby);
      lobbies.delete('RCN01');
    });

    it('updates lobby.hostId when host reconnects (Pitfall 1)', () => {
      const lobby = makeReconnectLobby('RCN02');
      reservePlayerSlot('RCN02', 'host-socket', () => {});
      assert.equal(lobby.hostId, 'host-socket'); // still old ID (not changed by reserve)
      const result = reconnectPlayer('RCN02', 'Alice', 'new-host-socket');
      assert.equal(result.ok, true);
      assert.equal(lobby.hostId, 'new-host-socket');
      const alice = lobby.players.find(p => p.name === 'Alice');
      assert.equal(alice.socketId, 'new-host-socket');
      cleanupTimers(lobby);
      lobbies.delete('RCN02');
    });

    it('returns error when no disconnected slot found', () => {
      makeReconnectLobby('RCN03');
      const result = reconnectPlayer('RCN03', 'Bob', 'new-socket');
      assert.equal(result.ok, false);
      assert.equal(result.error, 'Session expired');
      lobbies.delete('RCN03');
    });

    it('returns error for missing room', () => {
      const result = reconnectPlayer('NONEXISTENT', 'Bob', 'new-socket');
      assert.equal(result.ok, false);
      assert.equal(result.error, 'Room not found');
    });
  });

  describe('advanceTurn skips disconnected', () => {
    it('skips over disconnected player to next connected player', () => {
      const lobby = makeReconnectLobby('ADV01');
      // Disconnect Bob (index 1)
      reservePlayerSlot('ADV01', 'p2-socket', () => {});
      // Set turn to Alice (index 0), then advance
      lobby.activeTurnIndex = 0;
      advanceTurn(lobby);
      // Should skip Bob (disconnected at index 1) and land on Carol (index 2)
      assert.equal(lobby.activeTurnIndex, 2);
      cleanupTimers(lobby);
      lobbies.delete('ADV01');
    });

    it('handles all-disconnected without infinite loop', () => {
      const lobby = makeReconnectLobby('ADV02');
      reservePlayerSlot('ADV02', 'host-socket', () => {});
      reservePlayerSlot('ADV02', 'p2-socket', () => {});
      reservePlayerSlot('ADV02', 'p3-socket', () => {});
      // All disconnected — advanceTurn should complete without hanging
      lobby.activeTurnIndex = 0;
      advanceTurn(lobby); // must not infinite loop
      // Turn index can be anything — just verify it completed
      assert.ok(lobby.activeTurnIndex >= 0);
      cleanupTimers(lobby);
      lobbies.delete('ADV02');
    });
  });

  describe('getPublicState includes disconnected flag', () => {
    it('player objects contain disconnected field', () => {
      const lobby = makeReconnectLobby('GPS01');
      reservePlayerSlot('GPS01', 'p2-socket', () => {});
      const state = getPublicState('GPS01');
      const bob = state.players.find(p => p.name === 'Bob');
      assert.equal(bob.disconnected, true);
      const alice = state.players.find(p => p.name === 'Alice');
      assert.equal(alice.disconnected, false);
      cleanupTimers(lobby);
      lobbies.delete('GPS01');
    });
  });

  describe('host promotion on expiry', () => {

describe('Phase 15: reconnectRoom handler', () => {
  function makeMockSocket(socketId = 'mock-socket') {
    const handlers = {};
    const emitted = [];
    return {
      id: socketId,
      data: {},
      rooms: new Set(),
      on: (event, handler) => { handlers[event] = handler; },
      emit: (event, data) => { emitted.push({ event, data }); },
      to: () => ({ emit: (event, data) => { emitted.push({ event, data, broadcast: true }); } }),
      join: (room) => {},
      _handlers: handlers,
      _emitted: emitted,
    };
  }

  function makeMockIo() {
    const emitted = [];
    return {
      to: () => ({ emit: (event, data) => { emitted.push({ event, data }); } }),
      emit: (event, data) => { emitted.push({ event, data }); },
      _emitted: emitted,
    };
  }

  it('reconnectRoom emits game:stateUpdate on success', () => {
    const roomCode = 'SKREC01';
    lobbies.delete(roomCode);
    createLobby(roomCode, 'host-socket', 'Alice');
    addPlayer(roomCode, 'p2-socket', 'Bob');
    setSelectedPuzzle(roomCode, 'puzzle_01');
    startGame(roomCode);

    // Reserve Bob's slot (simulate disconnect)
    reservePlayerSlot(roomCode, 'p2-socket', () => {});

    const mockSocket = makeMockSocket('new-p2-socket');
    const mockIo = makeMockIo();
    const registerSocketHandlers = require('./socket');
    registerSocketHandlers(mockIo, mockSocket, new Map());

    // Trigger reconnectRoom handler
    mockSocket._handlers['reconnectRoom']({ roomCode, playerName: 'Bob' });

    // Should have emitted game:stateUpdate via io.to
    const stateUpdate = mockIo._emitted.find(e => e.event === 'game:stateUpdate');
    assert.ok(stateUpdate, 'Expected game:stateUpdate to be emitted');
    assert.equal(mockSocket.data.roomCode, roomCode);
    assert.equal(mockSocket.data.playerName, 'Bob');

    // Cleanup
    const lobby = getLobby(roomCode);
    if (lobby) lobby.players.forEach(p => { if (p.disconnectTimer) clearTimeout(p.disconnectTimer); });
    lobbies.delete(roomCode);
  });

  it('reconnectRoom emits room:error on failed reconnect', () => {
    const roomCode = 'SKREC02';
    lobbies.delete(roomCode);
    createLobby(roomCode, 'host-socket', 'Alice');

    registerSocketHandlers(mockIo, mockSocket, new Map());

    mockSocket._handlers['reconnectRoom']({ roomCode, playerName: 'Bob' });

    // Should have emitted room:error

    assert.ok(errorEmit, 'Expected room:error to be emitted');

    lobbies.delete(roomCode);
  });
});
