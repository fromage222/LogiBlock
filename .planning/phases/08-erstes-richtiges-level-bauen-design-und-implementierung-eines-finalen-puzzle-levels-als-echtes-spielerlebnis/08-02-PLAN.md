**Edit 2 — Update puzzle:list handler.** Replace the `option.textContent = p.name;` line so that the option text becomes "Name — Einfach" format when difficulty is present:

```javascript
// BEFORE:
socket.on('puzzle:list', (puzzles) => {
  puzzleSelect.innerHTML = '';
  puzzles.forEach(p => {
    const option = document.createElement('option');
    option.value = p.id;
    option.textContent = p.name;
    puzzleSelect.appendChild(option);
  });
});

// AFTER:
socket.on('puzzle:list', (puzzles) => {
  puzzleSelect.innerHTML = '';
  puzzles.forEach(p => {
    const option = document.createElement('option');
    option.value = p.id;
    const diffLabel = p.difficulty ? (DIFFICULTY_LABELS[p.difficulty] ?? p.difficulty) : '';
    option.textContent = diffLabel ? `${p.name} — ${diffLabel}` : p.name;
    puzzleSelect.appendChild(option);
  });
});
```
