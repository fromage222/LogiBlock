// ── Controls Modal (HLP-01) ─────────────────────────────────────────────────
controlsInfoBtn.addEventListener('click', () => {

controlsModalClose.addEventListener('click', () => {

controlsModal.addEventListener('click', (e) => {
  if (e.target === controlsModal) {
    controlsModal.close();
  }
});
